LeetCode.29 两数相除（mid）

给定两个整数，被除数 dividend 和除数 divisor。将两数相除，要求不使用乘法、除法和 mod 运算符。

返回被除数 dividend 除以除数 divisor 得到的商。

整数除法的结果应当截去（truncate）其小数部分，例如：truncate(8.345) = 8 以及 truncate(-2.7335) = -2

示例：

输入: dividend = 10, divisor = 3
输出: 3
解释: 10/3 = truncate(3.33333..) = truncate(3) = 3

常规做法

/**
 * @param {number} dividend
 * @param {number} divisor
 * @return {number}
 */
var divide = function (dividend, divisor) {
    if (dividend === -(2 ** 31) && divisor === -1) return 2 ** 31 - 1
    let res = 0
    let a = Math.abs(dividend)
    let b = Math.abs(divisor)
    while (a >= b) {
        a = a - b
        res++
    }
    if (dividend < 0 && divisor > 0 || dividend > 0 && divisor < 0) {
        res = -res
    }
    return res
};

位运算做法

var divide = function (dividend, divisor) {
    if (dividend === -(2 ** 31) && divisor === -1) return 2 ** 31 - 1
    let res = 0
    let a = Math.abs(dividend)
    let b = Math.abs(divisor)
    //核心代码，从大到小来，从31位开始试
    for (let i = 31; i >= 0; i--) { 
        if ((a >>> i) - b >= 0) {
            a -= (b << i)
            res += (1 << i)
        }
    }
    //注意底下这个if，是边界条件
    if (res === -(2 ** 31)) return res

    if (dividend < 0 && divisor > 0 || dividend > 0 && divisor < 0) {
        res = -res
    }
    return res
};

LeetCode.338 比特位计数 (easy)

给你一个整数 n ，对于 0 <= i <= n 中的每个 i ，计算其二进制表示中 1 的个数 ，返回一个长度为 n + 1 的数组 ans 作为答案。

示例：

输入：n = 2
输出：[0,1,1]
解释：
0 --> 0
1 --> 1
2 --> 10

常规做法

/**
 * @param {number} n
 * @return {number[]}
 */
var countBits = function (n) {
    let res = [0]
    for (let i = 1; i <= n; i++) {
        let binary = i.toString(2)
        let len = 0
        for (let j = 0; j < binary.length; j++) {
            if (binary[j] == 1) {
                len++
            }
        }
        res.push(len)
    }
    return res
};

位运算+dp

var countBits = function (n) {
    let res = [0]
    for (let i = 1; i <= n; i++) {
        res[i] = res[i - 1 & i] + 1
    }
    return res
};

LeetCode.136 只出现过一次的数字（easy）

给定一个非空整数数组，除了某个元素只出现一次以外，其余每个元素均出现两次。找出那个只出现了一次的元素。

示例：

输入: [4,1,2,1,2]
输出: 4

排序+遍历

var singleNumber = function (nums) {
    nums.sort()
    for (let i = 0; i < nums.length; i++) {
        if (nums[i] === nums[i + 1]) {
            i++
        } else {
            return nums[i]
        }
    }
};

利用map，不用排序，遍历两次

var singleNumber = function (nums) {
    const map = new Map()
    for (let i = 0; i < nums.length; i++) {
        map.set(nums[i], (map.get(nums[i]) || 0) + 1)
    }
    for(let i of map.keys()){
        if(map.get(i)===1){
            return i
        }
    }
};

位运算(异或)

//异或运算符能够使两个一样的数抵消成0 ，最后剩下的呢个就是出现过一次的元素
var singleNumber = function (nums) {
    let res = 0
    for (let i of nums) {
        res ^= i
    }
    return res
};

LeetCode.137 只出现过一次的数字(其他数字出现3次)（mid）

给你一个整数数组 nums ，除某个元素仅出现 一次 外，其余每个元素都恰出现 三次 。请你找出并返回那个只出现了一次的元素。

示例：

输入：nums = [0,1,0,1,0,1,100]
输出：100

利用map

var singleNumber = function (nums) {
    const map = new Map()
    for (let i of nums) {
        map.set(i, (map.get(i) || 0) + 1)
    }
    for (let i of map.keys()) {
        if (map.get(i) === 1) {
            return i
        }
    }
};

位运算

思路：出现三次的数，这个数的二进制对应位数之和就是3的倍数
比如： 15 8 15 15  1111 1000 1111 1111 对应二进制数之和为 4333 每个位对3取余 1000即为最后结果  
var singleNumber = function (nums) {
    let res = 0
    for (let i = 0; i < 32; i++) {
        let total = 0
        for (let num of nums) {
            total += (num >> i) & 1  //二进制从右往左走
        }
        if (total % 3 !== 0) {
            res |= (1 << i)
        }
    }
    return res
};

LeetCode.318 最大单词长度乘积（mid）

给你一个字符串数组 words ，找出并返回 length(words[i]) * length(words[j]) 的最大值，并且这两个单词不含有公共字母。如果不存在这样的两个单词，返回 0 。

示例：

输入：words = ["abcw","baz","foo","bar","xtfn","abcdef"]
输出：16 
解释：这两个单词为 "abcw", "xtfn"。

位运算+map

思路：利用map存储每个单词经过位运算操作后(实现了去除重复字母，二进制对应数为单词的ASCLL码之差)对应的十进制数与单词的最大长度
var maxProduct = function (words) {
    const map = new Map()
    let res = 0
    for (let word of words) {
        let m = 0
        for (let i of word) {
            m |= 1 << (i.charCodeAt() - 'a'.charCodeAt())
        }
        if (map.get(m)) {
            map.set(m, Math.max(map.get(m), word.length))
        } else {
            map.set(m, word.length)
        }
    }
    const arr = [...map.keys()]
    for (let i = 0; i < arr.length - 1; i++) {
        for (let j = i + 1; j < arr.length; j++) {
            if ((arr[i] & arr[j]) === 0) {
                res = Math.max(res, map.get(arr[i]) * map.get(arr[j]))
            }
        }
    }
    return res
};
